\section{代数}
\label{tricks-of-algebra}

\begin{example}
  证明:
  \begin{align*}
    \frac{a}{b+c} + \frac{b}{c+a} + \frac{c}{a+b} = 1 \quad \implies\quad
    \frac{a^2}{b+c} + \frac{b^2}{c+a} + \frac{c^2}{a+b} = 0
  \end{align*}
\end{example}
\begin{proof}[提示]
  两边同乘$a+b+c$，有
  \begin{align*}
    (a+b+c)\left(\frac{a}{b+c} + \frac{b}{c+a} + \frac{c}{a+b}\right) = a+b+c
  \end{align*}
  且若$a+b+c\ne 0$，则左右两式是等价的。
\end{proof}
